# function to find the most prevelant species per plot.
n_largest <- function(x = b, n) {
  
  # In descending order, prevelance
  x1 <- sort(x, decreasing = TRUE)
  # Select the nth most prevelant. 
  x1[n]
}
# function to remove empty columns. 
clean_zero_cols <- function(df) {
  x1 <- df |> select(!where(is.numeric))
  x2 <- df |> select(where(is.numeric))
  x2 <- x2[, colSums(x2) > ncol(x2)]
  df <- bind_cols(x1,x2)
  df
}
